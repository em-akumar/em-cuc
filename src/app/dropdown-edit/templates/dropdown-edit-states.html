        console.log('clicked');

    };

    

  


    
    

        

            

                Class/Properties
                Description
            

            

                .em-dropdown
                This class is required. Indicates a default/standard dropdown
            

             

                .disabled
                Make disabled state of dropdown control
            

            

                .readonly
                Make readonly state of dropdown control
            

            

                .restricted
                Make restricted state of dropdown control
            

            

                divider
                Property helps to add the divider inside the items ('true'/'false')
            

            

                .disabled
                Make disabled item by adding the disabled class e.g. itemList: [{ 'value': 1, 'text': 'State1', 'class': 'disabled' }
