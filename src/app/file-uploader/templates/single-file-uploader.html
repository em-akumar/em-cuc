Single File Uploader



File Uploader Usage
//Html usage
<div class="em-single-file-uploader">
  <div emc="SingleFileUpload" options="vm.singleFileoptions" ></div>
</div>
<button type="button" class="btn btn-primary" ng-click="vm.upload()">Upload</button>

//Controller usage
let options = {
      fileInputId: 'singleUpload',
      onselect: function(e){}
};
this.singleFileoptions = options;

upload(){
  if(document.getElementById('singleUpload').files.length> 0)
  {
    var file = document.getElementById('singleUpload').files[0];
    var fd = new FormData();
    fd.append('file', file);
    this.$http.post('http://10.112.104.13:85/v1/formbuilder/importpackage', fd, {
        transformRequest: angular.identity,
        headers: {'Content-Type': undefined}
    })
    .success(function(){
      console.log('file uploaded successfully.');
    })
    .error(function(){
      console.log('Error uploading file');
    });
  }
}
